泛类型命名规范
1、T: 通用类型
2、K/V： 键值对 （如map[K, V]）
3、IN/OUT : 输入/输出类型
4、E : 元素类型 （如切片，容器的元素）
5、comparable ： 可比较类型

[T ...] 中 T是类型变量
// 类型参数列表：[T 约束]，T是类型变量，约束限制T的范围
func 函数名[T 约束] (param []T) T {
    //通用逻辑
}

// 定义类型约束 允许 int /int64/ float64类型
type Number interface {
    int | int64 | float64 //类型集合， 竖线分割多个被允许的类型
}

func Sum[T Number] (nums []T) T {
    var sum T //类型
    for _, n := range nums {
        sum += n //通用逻辑，所有 Number 类型都支持 + 运算符
    }

    return  sum
}

// 结构体名[类型参数列表] 结构体字段
type 结构体名[T 约束] struct {
    Field []T
}

//结构体方法： 需要在方法名后声明类型参数
func (s *结构体名[T]) 方法名(param T) T {}

type Stack[T any] struct {
    elements []T
}

func (s *Stack[T]) Push(element T) {
    s.elements = append(s.elements, element)
